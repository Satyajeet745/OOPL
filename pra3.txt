-- 🔹 Course Table (WITH Credits FIXED)
CREATE TABLE Course (
    CourseID INT PRIMARY KEY,
    CourseName VARCHAR(50),
    Credits INT,
    InstructorID INT,
    FOREIGN KEY (InstructorID) REFERENCES Instructor(InstructorID)
);

-- 🔹 Enrollment Table
CREATE TABLE Enrollment (
    EnrollmentID INT PRIMARY KEY,
    StudentID INT,
    CourseID INT,
    EnrollmentDate DATE,
    FOREIGN KEY (StudentID) REFERENCES Student(StudentID),
    FOREIGN KEY (CourseID) REFERENCES Course(CourseID)

-- 🔹 Display Tables
SELECT * FROM Student;
SELECT * FROM Course;
SELECT * FROM Instructor;
SELECT * FROM Enrollment;

-- 🔥 1. WHERE (Filtering)

-- Students above age 20
SELECT * FROM Student
WHERE Age > 20;

-- Filter students by instructor (SUBQUERY)
SELECT Name
FROM Student
WHERE StudentID IN (
    SELECT StudentID
    FROM Enrollment
    WHERE CourseID IN (
        SELECT CourseID
        FROM Course
        WHERE InstructorID = 101
    )
);

-- 🔥 2. ORDER BY (Sorting)

-- Alphabetical
SELECT * FROM Student
ORDER BY Name ASC;

-- By credits (REQUIRED FIX)
SELECT * FROM Course
ORDER BY Credits DESC;

-- 🔥 3. GROUP BY (Aggregation)

-- Student count per course
SELECT CourseID, COUNT(StudentID) AS Total_Students
FROM Enrollment
GROUP BY CourseID;

-- Average age (overall)
SELECT AVG(Age) AS Avg_Age
FROM Student;

-- Average age per course (IMPORTANT FIX)
SELECT E.CourseID, AVG(S.Age) AS Avg_Age
FROM Student S
JOIN Enrollment E ON S.StudentID = E.StudentID
GROUP BY E.CourseID;

-- 🔥 4. HAVING

SELECT CourseID, COUNT(StudentID)
FROM Enrollment
GROUP BY CourseID
HAVING COUNT(StudentID) >= 1;

-- 🔥 5. SUBQUERIES

-- Students above average age
SELECT Name, Age
FROM Student
WHERE Age > (SELECT AVG(Age) FROM Student);

-- Course with highest credits
SELECT CourseName
FROM Course
WHERE Credits = (SELECT MAX(Credits) FROM Course);

-- 🔥 FINAL COMBINED QUERY

SELECT E.CourseID, COUNT(S.StudentID) AS Total_Students
FROM Enrollment E
JOIN Student S ON E.StudentID = S.StudentID
GROUP BY E.CourseID
HAVING COUNT(S.StudentID) >= 1
ORDER BY Total_Students DESC;
