CREATE DATABASE sales_management;
USE sales_management;

CREATE TABLE customer(
    customerid INT PRIMARY KEY,
    name VARCHAR(50),
    age INT
);

CREATE TABLE product(
    productid INT PRIMARY KEY,
    name VARCHAR(50),
    quantity INT,
    price DECIMAL(10,2)
);

CREATE TABLE orderss(
    orderid INT PRIMARY KEY,
    customerid INT
);

CREATE TABLE orderdetail(
    orderdetailid INT PRIMARY KEY,
    orderid INT,
    productid INT,
    quantity INT
);





DELIMITER //

CREATE PROCEDURE add_customer(
    IN p_customerid INT,
    IN p_name VARCHAR(50),
    IN p_age INT
)
BEGIN
    INSERT INTO customer VALUES(p_customerid,p_name,p_age);
END //

CREATE PROCEDURE add_product(
    IN p_productid INT,
    IN p_name VARCHAR(50),
    IN p_quantity INT,
    IN p_price DECIMAL(10,2)
)
BEGIN
    INSERT INTO product VALUES(p_productid,p_name,p_quantity,p_price);
END //

CREATE PROCEDURE add_orderss(
    IN p_orderid INT,
    IN p_customerid INT
)
BEGIN
    INSERT INTO orderss VALUES(p_orderid,p_customerid);
END //

CREATE PROCEDURE add_orderdetail(
    IN p_orderdetailid INT,
    IN p_orderid INT,
    IN p_productid INT,
    IN p_quantity INT
)
BEGIN
    INSERT INTO orderdetail VALUES(p_orderdetailid,p_orderid,p_productid,p_quantity);
END //

DELIMITER ;

DELIMITER //

CREATE FUNCTION total_sales()
RETURNS DECIMAL(12,2)
DETERMINISTIC
BEGIN
    DECLARE total DECIMAL(12,2);

    SELECT SUM(od.quantity * p.price)
    INTO total
    FROM orderdetail od, product p
    WHERE od.productid = p.productid;

    RETURN total;
END //

DELIMITER ;
